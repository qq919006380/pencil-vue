安装

注意

  • 请确保你的 Node.js 版本 >= 12。
npm i pencil-vue

注册组件



 


 


import { createApp } from 'vue'
import App from './App.vue'
import pencil from "pencil-vue";

var app = createApp(App)
app.use(pencil);
app.mount('#app')
